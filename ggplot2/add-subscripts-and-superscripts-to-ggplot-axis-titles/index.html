Add subscripts and superscripts to ggplot axis titles

Scott Prevost

2019/03/11

library(ggplot2)
library(scales)
library(lubridate)

df = data.frame(start_dt = seq(from = ymd_hms("2000-01-01 00:00:00"),
                               to = ymd_hms(" 2000-01-31 23:00:00"),
                               by = "1 hour"),
                temp = c(1:744))

plt = ggplot(data = df) +
  geom_line(aes(x = start_dt,
                y = temp)) + 
  scale_x_datetime(name = bquote("Date"*""[123]),
                   date_breaks = "1 week",
                   date_labels = "%b-%d") +
  scale_y_continuous(name = bquote("Temperature ("*""^o*"F)"),
                     labels = comma,
                     limits = c(0, 800),
                     breaks = seq(0, 800, 100))
